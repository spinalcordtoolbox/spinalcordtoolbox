70  void readImage(ImageType::Pointer image)
71  {
72  if (origin_[0] == 0.0 && origin_[1] == 0.0 && origin_[2] == 0.0) throw string("Error: reading image requires a defined origin");
73  else if (normal_[0] == 0.0 && normal_[1] == 0.0 && normal_[2] == 0.0) throw string("Error: reading image requires a defined origin");
74  else if (size_[0] == 0.0 && size_[1] == 0.0 && size_[2] == 0.0) throw string("Error: reading image requires a defined size");
75  else if (spacing_[0] == 0.0 && spacing_[1] == 0.0 && spacing_[2] == 0.0) throw string("Error: reading image requires a defined spacing");
76  else
77  {
78  if (content_ != 0) {
79  for (unsigned int x=0; x<size_[0]; x++) {
80  for (unsigned int y=0; y<size_[1]; y++)
81  delete [] content_[x][y];
82  delete [] content_[x];
83  }
84  delete [] content_;
85  }
86  content_ = new double**[size_[0]];
87  for (unsigned int x=0; x<size_[0]; x++)
88  {
89  content_[x] = new double*[size_[1]];
90  for (unsigned int y=0; y<size_[1]; y++)
91  content_[x][y] = new double[size_[2]];
92  }
93 
94  LinearInterpolatorType::Pointer imageInterpolator = LinearInterpolatorType::New();
95  imageInterpolator->SetInputImage(image);
96  ContinuousIndexType ind;
97 
98  ImageType::DirectionType directionImage = image->GetDirection();
99  CVector3 axe_AP_image = CVector3(*directionImage[1],*directionImage[4],*directionImage[7]).Normalize();
100  //cout << *directionImage[1] << " " << *directionImage[4] << " " << *directionImage[7] << endl;
101 
102  CVector3 normal(normal_[0],normal_[1],normal_[2]), directionCourantePerpendiculaire, origin(origin_[0],origin_[1],origin_[2]);
103  directionCourantePerpendiculaire = -(normal^axe_AP_image).Normalize();
104 
105  //if (normal[2] == 0.0) directionCourantePerpendiculaire = CVector3(0.0,0.0,1.0);
106  //else directionCourantePerpendiculaire = CVector3(1.0,2.0,-(normal[0]+2*normal[1])/normal[2]).Normalize();
107  refCourant_ = Referential(normal^directionCourantePerpendiculaire, directionCourantePerpendiculaire, normal, origin);
108 
109  CMatrix4x4 transformationFromOrigin = refCourant_.getTransformationInverse();
110  transformationFromOrigin[0] = spacing_[0]*transformationFromOrigin[0];
111  transformationFromOrigin[5] = spacing_[1]*transformationFromOrigin[5];
112  transformationFromOrigin[10] = spacing_[2]*transformationFromOrigin[10];
113  for (int x=0; x<size_[0]; x++)
114  {
115  for (int y=0; y<size_[1]; y++)
116  {
117  for (int z=0; z<size_[2]; z++)
118  {
119  PointType pt;
120  CVector3 ptInOrigin(x-(size_[0]-1)/2,y-(size_[1]-1)/2,z-(size_[2]-1)/2);
121  CVector3 ptInWorld = transformationFromOrigin*ptInOrigin;
122  pt[0] = ptInWorld[0]; pt[1] = ptInWorld[1]; pt[2] = ptInWorld[2];
123  //pt[0] = origin_[0]+normal_[0]*spacing_[0]*(x-(size_[0]-1)/2);
124  //pt[1] = origin_[1]+normal_[1]*spacing_[1]*(y-(size_[1]-1)/2);
125  //pt[2] = origin_[2]+normal_[2]*spacing_[2]*(z-(size_[2]-1)/2);
126  bool result = image->TransformPhysicalPointToContinuousIndex(pt,ind);
127  if (!result) throw string("Error: region of interest exceeds image dimension");
128  content_[x][y][z] = imageInterpolator->EvaluateAtContinuousIndex(ind);
129  }
130  }
131  }
132  }

328  float computeContrast(double& mean_distance, double& std_distance, int nbPoint=20)
329  {
330  InterpolateIntensityFilter::Pointer imageInterpolator = InterpolateIntensityFilter::New();
331  imageInterpolator->SetInputImage(image_);
332 
333  CVector3 centerImage = CVector3(((size_[0]-1)/2),((size_[1]-1)/2),((size_[2]-1)/2));
334 
335  mean_distance = 0.0;
336  std_distance = 0.0;
337  double rayon_ = 6.0;
338  vector<double> distance(nbPoint);
339  // Calcul du profil de la moelle et du LCR perpendiculairement au tube
340  itk::Point<double,3> pointS;
341  ContinuousIndexType indexS;
342  vector<float> contrast(nbPoint);
343  float angle;
344  CMatrix3x3 trZ;
345  for (int k=0; k<nbPoint; k++)
346  {
347  vector<float> profilIntensite;
348  angle = 2*M_PI*k/(double)nbPoint;
349  trZ[0] = cos(angle), trZ[1] = sin(angle), trZ[3] = -sin(angle), trZ[4] = cos(angle);
350  for (double l=0.0; l<=2.5*rayon_; l+=1) {
351  CVector3 pointS_temp = trZ*CVector3(l,0.0,0.0)+centerImage;
352  pointS[0] = pointS_temp[0]; pointS[1] = pointS_temp[1]; pointS[2] = pointS_temp[2];
353  if (image_->TransformPhysicalPointToContinuousIndex(pointS,indexS))
354  profilIntensite.push_back(factor_*imageInterpolator->EvaluateAtContinuousIndex(indexS));
355  }
356  float min = 0.0, max = 0.0, maxVal = 0.0, valCourante;
357  unsigned int m = 0;
358  for (unsigned int i=1; i<profilIntensite.size(); i++) {
359  valCourante = profilIntensite[i]-profilIntensite[i-1];
360  if (maxVal <= valCourante) {
361  maxVal = valCourante;
362  m = i;
363  }
364  }
365  if (profilIntensite.size() > 0)
366  {
367  min = profilIntensite[m];
368  for (unsigned int j=0; j<m; j++) {
369  valCourante = profilIntensite[j];
370  if (min > valCourante) min = valCourante;
371  }
372  max = profilIntensite[m];
373  for (unsigned int j=m+1; j<profilIntensite.size(); j++) {
374  valCourante = profilIntensite[j];
375  if (max < valCourante) max = valCourante;
376  }
377  }
378  contrast[k] = abs(max-min);
379  distance[k] = m;
380  }
381  float result = 0.0;
382  for (unsigned int i=0; i<contrast.size(); i++)
383  result += contrast[i];
384  result /= contrast.size();
385 
386  for (int n=0; n<nbPoint; n++) {
387  mean_distance += distance[n];
388  std_distance += distance[n]*distance[n];
389  }
390  mean_distance /= (double)nbPoint;
391  std_distance /= (double)nbPoint;
392  std_distance = sqrt(std_distance - mean_distance*mean_distance);
393  return result;
394  }

396  void build2DGaussian(double sigma)
397  {
398  image_ = ImageType::New();
399  ImageType::IndexType start; start.Fill(0);
400  ImageType::SizeType size; size[0] = size_[0]; size[1] = size_[1]; size[2] = size_[2];

406  image_->Allocate();
407  image_->FillBuffer(0);

409  itk::ImageRegionConstIterator<ImageType> imageIterator(image_,region);
410  ImageType::IndexType ind;
411  double value;
412  CVector3 centerImage = CVector3(((size_[0]-1)/2),((size_[1]-1)/2),((size_[2]-1)/2));
413  for (unsigned int x=0; x<size_[0]; x++)
414  {
415  for (unsigned int y=0; y<size_[1]; y++)
416  {
417  for (unsigned int z=0; z<size_[2]; z++)
418  {
419  ind[0] = x; ind[1] = y; ind[2] = z;
420  value = exp(-(x-centerImage[0])*(x-centerImage[0])/(2*sigma*sigma)-(y-centerImage[1])*(y-centerImage[1])/(2*sigma*sigma))/(2*M_PI*sigma*sigma);
421  image_->SetPixel(ind,value);
422  }
423  }
424  }

426  imageInterpolator->SetInputImage(image_);
