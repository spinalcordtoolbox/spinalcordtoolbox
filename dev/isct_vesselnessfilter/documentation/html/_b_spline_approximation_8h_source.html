37  BSplineApproximation(vector<CVector3>* centerline)
38  {
39  double range = centerline->size();
40 
41  PointSetType::Pointer pointSet = PointSetType::New();
42  // Sample the helix.
43  for (unsigned long i=0; i<range; i++) {
44  PointSetType::PointType point; point[0] = (double)i/(double)(range-1);
45  pointSet ->SetPoint( i, point );
46  VectorTypeSpline V;
47  V[0] = (*centerline)[i][0]; V[1] = (*centerline)[i][1]; V[2] = (*centerline)[i][2];
48  pointSet ->SetPointData( i, V );
49  }
50 
51  // Instantiate the filter and set the parameters
52  FilterTypeSpline::Pointer filter = FilterTypeSpline::New();
53  // Define the parametric domain
54  ImageTypeSpline::SpacingType spacing; spacing.Fill( 1.0 ); ImageTypeSpline::SizeType size; size.Fill( 2.0); ImageTypeSpline::PointType origin; origin.Fill( 0.0 );
55  ImageTypeSpline::RegionType region(size); FilterTypeSpline::ArrayType closedim; closedim.Fill(0);
56  filter->SetSize( size ); filter->SetOrigin( origin ); filter->SetSpacing( spacing ); filter->SetInput( pointSet );
57  int splineOrder = 3; filter->SetSplineOrder( splineOrder ); FilterTypeSpline::ArrayType ncps;
58  ncps.Fill( splineOrder + 1 ); filter->SetNumberOfControlPoints( ncps ); filter->SetNumberOfLevels( 5 ); filter->SetGenerateOutputImage( false );
59 
60  try
61  {
62  filter->Update();
63  } catch( itk::ExceptionObject & e ) {
64  std::cerr << "Exception caught while creating spline" << std::endl;
65  std::cerr << e << std::endl;
66  }
67 
68  bspline = BSplineType::New();
69  bspline->SetSplineOrder(filter->GetSplineOrder());
70  bspline->SetOrigin(origin);
71  bspline->SetSpacing(spacing);
72  bspline->SetSize(size);
73  bspline->SetInputImage(filter->GetPhiLattice());
74  };

76  CVector3 EvaluateBSpline(double value)
77  {
78  PointSetType::PointType point; point[0] = value;
79  VectorTypeSpline V = bspline->Evaluate( point );
80  return CVector3(V[0],V[1],V[2]);
81  };

86  BSplineType::GradientType Vd = bspline->EvaluateGradient( point );
87  return CVector3(Vd[0][0],Vd[1][0],Vd[2][0]);

90  double getNearestPoint(CVector3 point, double range)
91  {
92  double minIndex = 0;
93  double minDistance = 10000.0, distance = 0.0;
94  CVector3 point_bspline;
95  for (int k=0; k<range; k++)
96  {
97  point_bspline = EvaluateBSpline((double)k/range);
98  distance = sqrt((point[0]-point_bspline[0])*(point[0]-point_bspline[0])+(point[1]-point_bspline[1])*(point[1]-point_bspline[1])+(point[2]-point_bspline[2])*(point[2]-point_bspline[2]));
99  if (distance <= minDistance) {
100  minDistance = distance;
101  minIndex = k;
102  }
103  }
104  return (double)minIndex/range;
105  };
