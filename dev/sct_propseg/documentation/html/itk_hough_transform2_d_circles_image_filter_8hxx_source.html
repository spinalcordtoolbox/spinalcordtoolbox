78 template< typename TInputPixelType, typename TOutputPixelType >
79 void
80 HoughTransform2DCirclesImageFilter< TInputPixelType, TOutputPixelType >
81 ::GenerateData()
82 {
83  // Get the input and output pointers
84  InputImageConstPointer inputImage = this->GetInput(0);
85  OutputImagePointer outputImage = this->GetOutput(0);
86 
87  // Allocate the output
88  this->AllocateOutputs();
89  outputImage->FillBuffer(0);
90 
91  typedef GaussianDerivativeImageFunction< InputImageType > DoGFunctionType;
92  typename DoGFunctionType::Pointer DoGFunction = DoGFunctionType::New();
93  DoGFunction->SetInputImage(inputImage);
94  DoGFunction->SetSigma(m_SigmaGradient);
95 
96  m_RadiusImage = OutputImageType::New();
97 
98  m_RadiusImage->SetRegions( outputImage->GetLargestPossibleRegion() );
99  m_RadiusImage->SetOrigin( inputImage->GetOrigin() );
100  m_RadiusImage->SetSpacing( inputImage->GetSpacing() );
101  m_RadiusImage->SetDirection( inputImage->GetDirection() );
102  m_RadiusImage->Allocate();
103  m_RadiusImage->FillBuffer(0);
104 
105  ImageRegionConstIteratorWithIndex< InputImageType > image_it( inputImage, inputImage->GetRequestedRegion() );
106  image_it.GoToBegin();
107 
108  Index< 2 > index;
109  Point< float, 2 > point;
110 
111  while ( !image_it.IsAtEnd() )
112  {
113  if ( image_it.Get() > m_Threshold )
114  {
115  point[0] = image_it.GetIndex()[0];
116  point[1] = image_it.GetIndex()[1];
117  typename DoGFunctionType::VectorType grad = m_GradientFactor*DoGFunction->EvaluateAtIndex( image_it.GetIndex() );
118 
119  double Vx = grad[0];
120  double Vy = grad[1];
121 
122  // if the gradient is not flat
123  if ( ( vcl_fabs(Vx) > 1 ) || ( vcl_fabs(Vy) > 1 ) )
124  {
125  double norm = vcl_sqrt(Vx * Vx + Vy * Vy);
126  Vx /= norm;
127  Vy /= norm;
128 
129  for ( double angle = -m_SweepAngle; angle <= m_SweepAngle; angle += 0.05 )
130  {
131  double i = m_MinimumRadius;
132  double distance;
133 
134  do
135  {
136  index[0] = (IndexValueType)( point[0] - i * ( Vx * vcl_cos(angle) + Vy * vcl_sin(angle) ) );
137  index[1] = (IndexValueType)( point[1] - i * ( Vx * vcl_sin(angle) + Vy * vcl_cos(angle) ) );
138 
139  distance = vcl_sqrt( ( index[1] - point[1] ) * ( index[1] - point[1] )
140  + ( index[0] - point[0] ) * ( index[0] - point[0] ) );
141 
142  if ( outputImage->GetRequestedRegion().IsInside(index) )
143  {
144  outputImage->SetPixel(index, outputImage->GetPixel(index) + 1);
145  m_RadiusImage->SetPixel( index, ( m_RadiusImage->GetPixel(index) + distance ) );
146  }
147 
148  i = i + 1;
149  }
150  while ( outputImage->GetRequestedRegion().IsInside(index)
151  && ( distance < m_MaximumRadius ) );
152  }
153  }
154  }
155  ++image_it;
156  }
157 
158  // Compute the average radius
159  ImageRegionConstIterator< OutputImageType > output_it( outputImage, outputImage->GetLargestPossibleRegion() );
160  ImageRegionIterator< OutputImageType > radius_it( m_RadiusImage, m_RadiusImage->GetLargestPossibleRegion() );
161  output_it.GoToBegin();
162  radius_it.GoToBegin();
163  while ( !output_it.IsAtEnd() )
164  {
165  if ( output_it.Get() > 0 )
166  {
167  radius_it.Set( radius_it.Get() / output_it.Get() );
168  }
169  ++output_it;
170  ++radius_it;
171  }
172 }

175 template< typename TInputPixelType, typename TOutputPixelType >
176 typename HoughTransform2DCirclesImageFilter< TInputPixelType, TOutputPixelType >::CirclesListType &
177 HoughTransform2DCirclesImageFilter< TInputPixelType, TOutputPixelType >
178 ::GetCircles(unsigned int n)
179 {
180  if ( ( this->GetMTime() == m_OldModifiedTime ) && ( n == m_OldNumberOfCircles ) )
181  {
182  // if the filter has not been updated
183  return m_CirclesList;
184  }
185 
186  m_CirclesList.clear();
187 
189  typedef Image< float, 2 > InternalImageType;
190 
191  OutputImagePointer outputImage = OutputImageType::New();
192  outputImage->SetRegions( this->GetOutput(0)->GetLargestPossibleRegion() );
193  outputImage->SetOrigin( this->GetOutput(0)->GetOrigin() );
194  outputImage->SetSpacing( this->GetOutput(0)->GetSpacing() );
195  outputImage->SetDirection( this->GetOutput(0)->GetDirection() );
196  outputImage->Allocate();
197  outputImage->FillBuffer(0);
198 
199  ImageRegionConstIteratorWithIndex< OutputImageType > image_it( this->GetOutput(0), this->GetOutput(
200  0)->GetRequestedRegion() );
201  image_it.GoToBegin();
202 
203  ImageRegionIterator< OutputImageType > it( outputImage, outputImage->GetRequestedRegion() );
204 
205  while ( !image_it.IsAtEnd() )
206  {
207  it.Set( image_it.Get() );
208  ++image_it;
209  ++it;
210  }

212  typedef DiscreteGaussianImageFilter< OutputImageType, InternalImageType > GaussianFilterType;
213  typename GaussianFilterType::Pointer gaussianFilter = GaussianFilterType::New();
214 
215  gaussianFilter->SetInput(outputImage); // the output is the accumulator image
216  double variance[2];
217  variance[0] = m_Variance;
218  variance[1] = m_Variance;
219  gaussianFilter->SetVariance(variance);
220  gaussianFilter->Update();
221  typename InternalImageType::Pointer postProcessImage = gaussianFilter->GetOutput();
222 
223  typedef MinimumMaximumImageCalculator< InternalImageType > MinMaxCalculatorType;
224  typename MinMaxCalculatorType::Pointer minMaxCalculator = MinMaxCalculatorType::New();
225  ImageRegionIterator< InternalImageType > it_input( postProcessImage, postProcessImage->GetLargestPossibleRegion() );
226 
227  Index< 2 > index;
228 
229  CirclesListSizeType circles = 0;
230  bool found;
231 
232  const double nPI = 4.0 * vcl_atan(1.0);
233 
234  // Find maxima
235  do
236  {
237  minMaxCalculator->SetImage(postProcessImage);
238  minMaxCalculator->ComputeMaximum();
239  InternalImageType::PixelType max = minMaxCalculator->GetMaximum();
240 
241  found = false;
242  for ( it_input.GoToBegin(); !it_input.IsAtEnd(); ++it_input )
243  {
244  if ( it_input.Get() == max )
245  {
246  // Create a Line Spatial Object
247  CirclePointer Circle = CircleType::New();
248  Circle->SetId(circles);
249  Circle->SetRadius( m_RadiusImage->GetPixel( it_input.GetIndex() ) );
250 
251  CircleType::VectorType center;
252  center[0] = it_input.GetIndex()[0];
253  center[1] = it_input.GetIndex()[1];
254  Circle->GetObjectToParentTransform()->SetOffset(center);
255  Circle->ComputeBoundingBox();
256 
257  m_CirclesList.push_back(Circle);
258 
259  // Remove a black disc from the hough space domain
260  for ( double angle = 0; angle <= 2 * nPI; angle += nPI / 1000 )
261  {
262  for ( double length = 0; length < m_DiscRadiusRatio * Circle->GetRadius()[0]; length += 1 )
263  {
264  index[0] = (IndexValueType)( it_input.GetIndex()[0] + length * vcl_cos(angle) );
265  index[1] = (IndexValueType)( it_input.GetIndex()[1] + length * vcl_sin(angle) );
266  if ( postProcessImage->GetLargestPossibleRegion().IsInside(index) )
267  {
268  postProcessImage->SetPixel(index, 0);
269  }
270  }
271  }
272  minMaxCalculator->SetImage(postProcessImage);
273  minMaxCalculator->ComputeMaximum();
274  max = minMaxCalculator->GetMaximum();
275 
276  circles++;
277  found = true;
278  if ( circles == m_NumberOfCircles ) { break; }
279  }
280  }
281  }
282  while ( ( circles < m_NumberOfCircles ) && ( found ) );
283 
284  m_OldModifiedTime = this->GetMTime();
285  m_OldNumberOfCircles = m_CirclesList.size();
286  return m_CirclesList;
287 }
